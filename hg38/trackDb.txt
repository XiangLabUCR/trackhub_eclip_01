#######################
# PB1 composite track
#######################
track PB1
compositeTrack on
shortLabel PB1 eCLIP
longLabel PB1 eCLIP IP and Input
group default
visibility full
type bigWig
autoScale on
maxHeightPixels 128:32:8
subGroup1 sample IP=IP IN=Input
subGroup2 replicate 1=Rep1 2=Rep2
sortOrder sample=+ replicate=+
dimensions dimensionX=replicate
dragAndDrop subTracks
priority 1

    # ---- PB1 IN Rep1 ----
    track PB1_IN_1
    parent PB1 on
    bigDataUrl PB1_IN_1.bw
    shortLabel IN_1
    longLabel PB1 Input replicate 1
    subGroups sample=IN replicate=1
    color 0,100,255
    negateValues on

    # ---- PB1 IN Rep2 ----
    track PB1_IN_2
    parent PB1 on
    bigDataUrl PB1_IN_2.bw
    shortLabel IN_2
    longLabel PB1 Input replicate 2
    subGroups sample=IN replicate=2
    color 0,100,255

    # ---- PB1 IP Rep1 ----
    track PB1_IP_1
    parent PB1 on
    bigDataUrl PB1_IP_1.bw
    shortLabel IP_1
    longLabel PB1 IP replicate 1
    subGroups sample=IP replicate=1
    color 255,0,0

    # ---- PB1 IP Rep2 ----
    track PB1_IP_2
    parent PB1 on
    bigDataUrl PB1_IP_2.bw
    shortLabel IP_2
    longLabel PB1 IP replicate 2
    subGroups sample=IP replicate=2
    color 255,0,0

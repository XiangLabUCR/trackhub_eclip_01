#######################
# PB1 composite track
#######################
track PB1
compositeTrack on
shortLabel PB1 eCLIP
longLabel PB1 eCLIP IP and Input
group default
visibility full
type bigWig
autoScale on
maxHeightPixels 128:32:8
subGroup1 sample IP=IP IN=Input
subGroup2 replicate 1=Rep1 2=Rep2
sortOrder sample=+ replicate=+
dimensions dimensionX=replicate
dragAndDrop subTracks
priority 1

    # ---- PB1 IN Rep1 ----
    track PB1_IN_1
    parent PB1 on
    view Signal
    bigDataUrl PB1_IN_1.bw
    shortLabel IN_1
    longLabel PB1 Input replicate 1
    subGroups sample=IN replicate=1
    color 0,100,255
    negateValues on

    # ---- PB1 IN Rep2 ----
    track PB1_IN_2
    parent PB1 on
    view Signal
    bigDataUrl PB1_IN_2.bw
    shortLabel IN_2
    longLabel PB1 Input replicate 2
    subGroups sample=IN replicate=2
    color 0,100,255

    # ---- PB1 IP Rep1 ----
    track PB1_IP_1
    parent PB1
    view Signal
    shortLabel IP1
    longLabel PB1 IP replicate 1
    type bigWig

        track PB1_IP_1_plus
        parent PB1_IP_1
        shortLabel +
        longLabel PB1 IP1 plus strand
        bigDataUrl PB1_IP_1.scaled.plus.bw
        color 0,102,204

        track PB1_IP_1_minus
        parent PB1_IP_1
        shortLabel -
        longLabel PB1 IP1 minus strand
        bigDataUrl PB1_IP_1.scaled.minus.bw
        color 204,0,0
        negateValues on

    # ---- PB1 IP Rep2 ----
    track PB1_IP_2
    parent PB1
    view Signal
    shortLabel IP2
    longLabel PB1 IP replicate 2
    type bigWig

        track PB1_IP_2_plus
